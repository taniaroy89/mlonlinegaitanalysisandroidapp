	\tiny{
	HMM $= <N, M, \mathbf{A}, \mathbf{B}, \mathbf{\pi}>$
dove: 

\begin{enumerate}
	\item $N = |S|$: stati nascosti
	\item $M = |V|$:  alfabeto osservazione
	\item $\mathbf{A}\{a_{ij}\}$: probabilità di transizione 
				$a_{ij} = \wp[q_t = S_j | q_{t-1} = S_i]$
	\item $\mathbf{B}$ :probabilità di emissione
				$b_j(k) = \wp[v_k \; \text{all'istante } t | q_t = s_j] $
	\item $\mathbf{\pi}$: probabilità a priori
				$\pi_i = \wp(q_1 = S_i)$
				
\end{enumerate}
